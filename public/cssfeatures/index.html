Main Page Styles

.content {
    display: grid;
    grid-template-columns: 1fr 3fr;
    grid-template-rows: auto 1fr;
    grid-template-areas:
        "header header"
        "sidebar main";

    & .sidebar {
        grid-area: sidebar;
    }

    & .header {
        grid-area: header;
        text-align: center;
    }

    & .main {
        grid-area: main;
    }
}
